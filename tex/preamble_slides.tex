        {\scshape \usebeamerfont{title}\usebeamercolor[fg]{title}\inserttitle}\\
        {\color{accent} \rule{10cm}{0.3pt}}\\[5mm]

        {\setlength{\columnsep}{-100pt}   
        \begin{multicols}{\authorcolumns}
            \insertauthor
        \end{multicols}
        }

        \vspace{-2em}

        {\setlength{\columnsep}{-100pt}
        \begin{multicols}{\authorcolumns}
            {\color{jet}\small\insertinstitute}
        \end{multicols}
        }

        \vspace{5mm}
        {\color{asher}\small\insertdate}\\
    \end{centering}
}
